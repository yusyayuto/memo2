Sub シート名の括弧内の数字で並び替え()
    Dim i As Integer, j As Integer
    Dim sheetCount As Integer
    Dim sheetInfo() As Variant

    sheetCount = Sheets.Count
    ReDim sheetInfo(1 To sheetCount, 1 To 2)

    ' シート名と括弧内の数字を取得
    For i = 1 To sheetCount
        With Sheets(i)
            sheetInfo(i, 1) = .Name
            sheetInfo(i, 2) = GetNumberFromParentheses(.Name)
        End With
    Next i

    ' 昇順でバブルソート
    For i = 1 To sheetCount - 1
        For j = i + 1 To sheetCount
            If sheetInfo(i, 2) > sheetInfo(j, 2) Then
                Dim tmp1 As Variant, tmp2 As Variant
                tmp1 = sheetInfo(i, 1)
                tmp2 = sheetInfo(i, 2)
                sheetInfo(i, 1) = sheetInfo(j, 1)
                sheetInfo(i, 2) = sheetInfo(j, 2)
                sheetInfo(j, 1) = tmp1
                sheetInfo(j, 2) = tmp2
            End If
        Next j
    Next i

    ' 並び替え実行
    For i = sheetCount To 1 Step -1
        Sheets(sheetInfo(i, 1)).Move Before:=Sheets(1)
    Next i
End Sub

' 括弧内の数字を抽出（全角・半角どちらも対応）
Function GetNumberFromParentheses(sheetName As String) As Long
    Dim regex As Object
    Set regex = CreateObject("VBScript.RegExp")

    With regex
        .Pattern = "[\(\（](\d+)[\)\）]"
        .Global = False
        If .Test(sheetName) Then
            GetNumberFromParentheses = CLng(.Execute(sheetName)(0).SubMatches(0))
        Else
            GetNumberFromParentheses = 0
        End If
    End With
End Function
